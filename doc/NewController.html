def index
  @page = normalized_page

  ids = HackerNews::Request.new.get(api: :hn, resource: :new_stories)[:data].slice(page_range)

  @stories = get_items_from_ids(ids)

  @stories.each_with_index do |story, i|
    set_story_rank_link_score(story, i)
  end
end
