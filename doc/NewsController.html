def index
  @page = normalized_page

  ids = HackerNews::Request.new.get(api: :hn, resource: :top_stories)[:data].slice(page_range)

  @items = get_items_from_ids(ids)

  @items.each_with_index do |item, i|
    set_item_rank_link_score(item, i)
  end
end
